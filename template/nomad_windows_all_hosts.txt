	constraint {
		attribute = "${attr.kernel.name}"
		value     = "windows"
	}

		operator = "distinct_hosts"
		value = "true"
	}
	group "windowsupload" {
		restart {
			attempts = 0
			mode     = "fail"
		}

			driver = "raw_exec"

				cpu    = 100
				memory = 20
			}
			logs {
				max_files     = 10
				max_file_size = 10
			}
			config {
				command = "powershell"
				args = ["{{ cmd }}",{{ args }}]
			}
		}
	}
}
